7.b) Write a computer program to find the solution of a simple harmonic oscillator (no friction) using RK2 method, given $k=1$ and plot it using matplotlib module.¶

Download this program

In [15]:
import numpy as np 
import matplotlib.pyplot as plt 
def f(t,x,dxdt):
    return -k*x
t,x,dxdt=0,1,0
k=1
t_final,dt=25,0.01
T,V,A=[],[],[]
while t<=t_final:
    a1=dt*dxdt
    b1=dt*f(t,x,dxdt)
    a2=dt*(dxdt+0.5*b1)
    b2=dt*f(t+0.5*dt,x+0.5*a1,dxdt+0.5*b1)
    dx=0.5*(a1+a2)
    x+=dx
    dxdt+=0.5*(b1+b2)
    d2xdt2=dx/dt
    T.append(t)
    V.append(dxdt)
    A.append(d2xdt2)
    t+=dt
plt.plot(T,A,label='for k=1')
plt.legend(loc='best',prop={'size':12})
plt.xlabel('Time(t)')
plt.ylabel('Position(x)')
plt.title('Motion of Simple Harmonic Oscillator(no friction)')
plt.show()

8) Solve the ODE problem¶

$\frac{dT}{dt}=-\frac{1}{27}(T-65),T(0)=200°F$

Using the Euler method in the range [0.0,10.0] with step 1.0. Plot the numerical solution together with the exact solution $T(t)=65+135e^{-t}/27$¶

Download this program

In [16]:
import matplotlib.pyplot as plt 
from math import exp
def f(t,T):
    return -(T-65)/27
def g(t):
    return 65+(135*exp(-t))/27

t,T,h=0,200,1.0
time,temp,ex_temp=[],[],[]
n=0
while n<10:
    T+=h*f(t,T)
    t+=h
    time.append(t)
    temp.append(T)
    ex_temp.append(g(t))
    n+=h
plt.plot(time,temp,label='solution by Euler method')
plt.plot(time,ex_temp,label='exact solution')
plt.legend(loc='best',prop={'size':12})
plt.xlabel('t')
plt.ylabel('T')
plt.title('Numerical solution of given ODE')
plt.show()
